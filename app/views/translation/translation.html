                            <!-- Number should be rounded using locale -->
                            <p>{{1200300.456789 | number}}</p>

                            <h4>Text</h4>

                            <p translate>WELCOME</p>

                            <!-- With values -->
                            <p translate translate-values='{"name": "Alex"}'>WELCOME_NAME</p>

                            <!-- Using scope variables -->
                            <p translate translate-values='{"name": "{{name}}", "age": {{age}} }'>WELCOME_NAME_AGE</p>
                            </div>
                        
				
                        
.controller('translationController', ['$scope', '$translate', 'tmhDynamicLocale',
     function ($scope, $translate, tmhDynamicLocale) {
        $scope.now = new Date();
        $scope.name = 'Alicia';
        $scope.age = '25';
        $scope.setLocale = function (locale) {
            $translate.uses(locale);
            tmhDynamicLocale.set(locale.toLowerCase());
        };
      }]);
